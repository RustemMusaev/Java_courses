CREATE TABLE citys
(
  id serial NOT NULL,
  name character varying,
  area double precision,
  count_people integer,
  CONSTRAINT fk_citys PRIMARY KEY (id)
)
CREATE TABLE organizations
(
  id serial NOT NULL,
  name character varying,
  city_id integer,
  street_id integer,
  house_number integer,
  description character varying,
  website character varying,
  date_update timestamp with time zone,
  CONSTRAINT fk_organizations PRIMARY KEY (id),
  CONSTRAINT pk_organization_citys FOREIGN KEY (city_id)
      REFERENCES citys (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT pk_organizations_sreets FOREIGN KEY (street_id)
      REFERENCES streets (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
CREATE TABLE streets
(
  id serial NOT NULL,
  name character varying,
  length integer,
  CONSTRAINT fk_streets PRIMARY KEY (id)
)
CREATE TABLE phones
(
  id serial NOT NULL,
  "number" character varying,
  organization_id integer,
  CONSTRAINT fk_phones PRIMARY KEY (id),
  CONSTRAINT phones_organization_id_fkey FOREIGN KEY (organization_id)
      REFERENCES organizations (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
